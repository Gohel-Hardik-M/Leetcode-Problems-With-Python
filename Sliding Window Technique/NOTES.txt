âœ… Example (Fixed Size)
Problem: Maximum sum of any subarray of size k.

CODE EXAMPLE :

def max_sum_k(nums, k):
    window_sum = sum(nums[:k])
    max_sum = window_sum

    for i in range(k, len(nums)):
        window_sum += nums[i] - nums[i - k]   # slide window
        max_sum = max(max_sum, window_sum)

    return max_sum

# Example
print(max_sum_k([2, 1, 5, 1, 3, 2], 3))  # Output: 9

def longest_unique_substring(s):
    seen = set()
    left = 0
    max_len = 0

    for right in range(len(s)):
        while s[right] in seen:
            seen.remove(s[left])
            left += 1

        seen.add(s[right])
        max_len = max(max_len, right - left + 1)

    return max_len

# Example
print(longest_unique_substring("abcabcbb"))  # Output: 3
